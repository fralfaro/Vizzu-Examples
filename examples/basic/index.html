    import pandas as pd
from ipyvizzu import Chart, Data, Config, DisplayTarget

# Cargar los datos del Titanic desde un archivo CSV
df = pd.read_csv("https://ipyvizzu.vizzuhq.com/latest/showcases/titanic/titanic.csv")

# Crear un objeto Data y añadir el dataframe
data = Data()
data.add_df(df)

# Crear un objeto Chart con el ancho y el alto deseados 
chart = Chart(width="640px", height="360px",display=DisplayTarget.END)

# Animar el gráfico con los datos
chart.animate(data)

# Animar el gráfico con la primera configuración: un gráfico de barras que muestra el conteo de pasajeros por sexo
chart.animate(
    Config(
        {
            "x": "Count",
            "y": "Sex",
            "label": "Count",
            "title": "Passengers of the Titanic",
        }
    )
)

# Animar el gráfico con la segunda configuración: un gráfico de barras apiladas que muestra el conteo y la supervivencia de los pasajeros por sexo
chart.animate(
    Config(
        {
            "x": ["Count", "Survived"],
            "label": ["Count", "Survived"],
            "color": "Survived",
        }
    )
)

# Animar el gráfico con la tercera configuración: un gráfico de barras agrupadas que muestra el conteo de los pasajeros por sexo y supervivencia
chart.animate(Config({"x": "Count", "y": ["Sex", "Survived"]}))
